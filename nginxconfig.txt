user nginx;
worker_processes auto;
error_log /var/log/nginx/error.log;
pid /run/nginx.pid;

# Load dynamic modules. See /usr/share/doc/nginx/README.dynamic.
include /usr/share/nginx/modules/*.conf;

events {
    worker_connections 1024;
}

http {
    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;

    sendfile            on;
    tcp_nopush          on;
    tcp_nodelay         on;
    keepalive_timeout   65;
    types_hash_max_size 2048;
    server_names_hash_bucket_size 64;

    include             /etc/nginx/mime.types;
    default_type        application/octet-stream;

    # Load modular configuration files from the /etc/nginx/conf.d directory.
    # See http://nginx.org/en/docs/ngx_core_module.html#include
    # for more information.
    include /etc/nginx/conf.d/*.conf;

    upstream web1 {
        server 166.16.0.26 weight=1; #此处填写被代理服务器IP
        server 166.16.0.55 weight=1; #此处填写被代理服务器IP
        #ip_hash; #登录重定向
    }

    server {
        listen 80;
        server_name www.xiba.com; #此处填写当前网站域名
        access_log  /var/log/nginx/farmsec-org.log; #此处填写反向代理服务器日志位置
        location / {
            root /home/web1_root;
            proxy_pass http://web1; #与upstream一致
            proxy_read_timeout 300;
            proxy_connect_timeout 300;
            proxy_redirect     off;
            proxy_set_header   X-Forwarded-Proto $scheme;
            proxy_set_header   Host              $http_host;
            proxy_set_header   X-Real-IP         $remote_addr;
            proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
        }
    }


    upstream web2 {
        server 166.16.0.48 weight=1; #此处填写被代理服务器IP
        server 166.16.0.36 weight=1; #此处填写被代理服务器IP
        #ip_hash; #登录重定向
    }

    server {
        listen 80;
        server_name www.xiba2.com; #此处填写当前网站域名
        access_log  /var/log/nginx/farmsec-io.log; #此处填写反向代理服务器日志位置
        location / {
            root /home/web2_root;
            proxy_pass http://web2; #与upstream一致
            proxy_read_timeout 300;
            proxy_connect_timeout 300;
            proxy_redirect     off;
            proxy_set_header   X-Forwarded-Proto $scheme;
            proxy_set_header   Host              $http_host;
            proxy_set_header   X-Real-IP         $remote_addr;
            proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
        }
    }
